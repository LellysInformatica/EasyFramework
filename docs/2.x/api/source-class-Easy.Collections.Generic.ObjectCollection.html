  3: /**
  4:  * Deals with Collections of objects.
  5:  *
  6:  * Keeping registries of those objects,
  7:  * loading and constructing new objects and triggering callbacks. Each subclass needs
  8:  * to implement its own load() functionality.
  9:  *
 10:  * All core subclasses of ObjectCollection by convention loaded objects are stored
 11:  * in `$this->_loaded`. Enabled objects are stored in `$this->_enabled`. In addition
 12:  * the all support an `enabled` option that controls the enabled/disabled state of the object
 13:  * when loaded.
 14:  *
 15:  * @since EasyFW v 0.3
 16:  */
 17: 
 18: namespace Easy\Collections\Generic;
 19: 
 20: use Easy\Collections\Dictionary;
 21: use Easy\Event\Event;
 22: 
 23: /**
 24:  * Manage Objects collections
 25:  * 
 26:  * @package Easy.Collections.Generic
 27:  */
 28: abstract class ObjectCollection extends Dictionary
 29: {
 30: 
 31:     /**
 32:      * Loads a new object onto the collection.
 33:      * Can throw a variety of exceptions
 34:      *
 35:      * Implementations of this class support a `$options['enabled']` flag which enables/disables
 36:      * a loaded object.
 37:      *
 38:      * @param $name string Name of object to load.
 39:      * @param $options array Array of configuration options for the object to be constructed.
 40:      * @return object the constructed object
 41:      */
 42:     abstract public function load($name, $options = array());
 43: 
 44:     /**
 45:      * Trigger a callback method on every object in the collection.
 46:      * Used to trigger methods on objects in the collection. Will fire the methods in the
 47:      * order they were attached.
 48:      *
 49:      * ### Options
 50:      *
 51:      * - `breakOn` Set to the value or values you want the callback propagation to stop on.
 52:      * Can either be a scalar value, or an array of values to break on. Defaults to `false`.
 53:      *
 54:      * - `break` Set to true to enabled breaking. When a trigger is broken, the last returned value
 55:      * will be returned. If used in combination with `collectReturn` the collected results will be
 56:      * returned.
 57:      * Defaults to `false`.
 58:      *
 59:      * - `collectReturn` Set to true to collect the return of each object into an array.
 60:      * This array of return values will be returned from the trigger() call. Defaults to `false`.
 61:      *
 62:      * - `modParams` Allows each object the callback gets called on to modify the parameters to the
 63:      * next object.
 64:      * Setting modParams to an integer value will allow you to modify the parameter with that index.
 65:      * Any non-null value will modify the parameter index indicated.
 66:      * Defaults to false.
 67:      *
 68:      *
 69:      * @param obj string
 70:      *        Method to fire on all the objects. Its assumed all the objects implement
 71:      *        the method you are calling.
 72:      * @param objrray
 73:      *        Array of parameters for the triggered callback.
 74:      * @param objarray
 75:      *        Array of options.
 76:      * @return mixed Either the last result or all results if collectReturn is on.
 77:      * @throws CakeException when modParams is used with an index that does not exist.
 78:      */
 79:     public function trigger($callback, $params = array(), $options = array())
 80:     {
 81:         if ($callback instanceof Event) {
 82:             $event = $callback;
 83:             if (is_array($event->data)) {
 84:                 $params = &$event->data;
 85:             }
 86:             if (empty($event->omitSubject)) {
 87:                 $subject = $event->subject();
 88:                 $params = &$subject;
 89:             }
 90:             //TODO: Temporary BC check, while we move all the triggers system into the CakeEventManager
 91:             foreach (array('break', 'breakOn', 'collectReturn', 'modParams') as $opt) {
 92:                 if (isset($event->{$opt})) {
 93:                     $options[$opt] = $event->{$opt};
 94:                 }
 95:             }
 96:             $parts = explode('.', $event->name());
 97:             $callback = array_pop($parts);
 98:         }
 99: 
100:         $options = array_merge(array(
101:             'break' => false,
102:             'breakOn' => false,
103:             'collectReturn' => false,
104:             'modParams' => false), $options);
105: 
106:         foreach ($this->GetArray() as $obj) {
107:             if (method_exists($obj, $callback)) {
108:                 $obj->{$callback}($params);
109:             } else {
110:                 trigger_error("O método {$callback} não pode ser chamado na classe {$obj}", E_USER_WARNING);
111:             }
112:         }
113:     }
114: 
115:     /**
116:      * Normalizes an object array, creates an array that makes lazy loading
117:      * easier
118:      *
119:      * @param $objects array Array of child objects to normalize.
120:      * @return array Array of normalized objects.
121:      */
122:     public static function normalizeObjectArray($objects)
123:     {
124:         $normal = array();
125:         foreach ($objects as $i => $objectName) {
126:             $options = array();
127:             if (!is_int($i)) {
128:                 $options = (array) $objectName;
129:                 $objectName = $i;
130:             }
131:             $normal [$name] = array('class' => $objectName,
132:                 'settings' => $options);
133:         }
134:         return $normal;
135:     }
136: 
137:     /**
138:      * Provide public read access to the loaded objects
139:      *
140:      * @param string $name Name of property to read
141:      * @return mixed
142:      */
143:     public function __get($name)
144:     {
145:         if (isset($this->array[$name])) {
146:             return $this->array[$name];
147:         }
148:         return null;
149:     }
150: 
151:     /**
152:      * Provide isset access to _loaded
153:      *
154:      * @param string $name Name of object being checked.
155:      * @return boolean
156:      */
157:     public function __isset($name)
158:     {
159:         return isset($this->array[$name]);
160:     }
161: 
162: }
