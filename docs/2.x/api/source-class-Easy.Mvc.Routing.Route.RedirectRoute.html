 23: /**
 24:  * Redirect route will perform an immediate redirect. Redirect routes
 25:  * are useful when you want to have Routing layer redirects occur in your
 26:  * application, for when URLs move.
 27:  *
 28:  * @package Easy.Routing.Route
 29:  */
 30: class RedirectRoute extends Route
 31: {
 32: 
 33:     /**
 34:      * A CakeResponse object
 35:      *
 36:      * @var CakeResponse
 37:      */
 38:     public $response = null;
 39: 
 40:     /**
 41:      * The location to redirect to. Either a string or a cake array url.
 42:      *
 43:      * @var mixed
 44:      */
 45:     public $redirect;
 46: 
 47:     /**
 48:      * Flag for disabling exit() when this route parses a url.
 49:      *
 50:      * @var boolean
 51:      */
 52:     public $stop = true;
 53: 
 54:     /**
 55:      * Constructor
 56:      *
 57:      * @param string $template Template string with parameter placeholders
 58:      * @param array $defaults Array of defaults for the route.
 59:      * @param array $options Array of additional options for the Route
 60:      */
 61:     public function __construct($template, $defaults = array(), $options = array())
 62:     {
 63:         parent::__construct($template, $defaults, $options);
 64:         $this->redirect = (array) $defaults;
 65:     }
 66: 
 67:     /**
 68:      * Parses a string url into an array. Parsed urls will result in an automatic
 69:      * redirection
 70:      *
 71:      * @param string $url The url to parse
 72:      * @return boolean False on failure
 73:      */
 74:     public function parse($url)
 75:     {
 76:         $params = parent::parse($url);
 77:         if (!$params) {
 78:             return false;
 79:         }
 80:         if (!$this->response) {
 81:             $this->response = new Response();
 82:         }
 83:         $redirect = $this->redirect;
 84:         if (count($this->redirect) == 1 && !isset($this->redirect['controller'])) {
 85:             $redirect = $this->redirect[0];
 86:         }
 87:         if (isset($this->options['persist']) && is_array($redirect)) {
 88:             $redirect += array('named' => $params['named'], 'pass' => $params['pass'], 'url' => array());
 89:             $redirect = Router::reverse($redirect);
 90:         }
 91:         $status = 301;
 92:         if (isset($this->options['status']) && ($this->options['status'] >= 300 && $this->options['status'] < 400)) {
 93:             $status = $this->options['status'];
 94:         }
 95:         $this->response->header(array('Location' => Router::url($redirect, true)));
 96:         $this->response->setStatusCode($status);
 97:         $this->response->send();
 98:         $this->_stop();
 99:     }
100: 
101:     /**
102:      * There is no reverse routing redirection routes
103:      *
104:      * @param array $url Array of parameters to convert to a string.
105:      * @return mixed either false or a string url.
106:      */
107:     public function match($url)
108:     {
109:         return false;
110:     }
111: 
112:     /**
113:      * Stop execution of the current script. Wraps exit() making
114:      * testing easier.
115:      *
116:      * @param integer|string $status see http://php.net/exit for values
117:      * @return void
118:      */
119:     protected function _stop($code = 0)
120:     {
121:         if ($this->stop) {
122:             exit($code);
123:         }
124:     }
125: 
126: }
