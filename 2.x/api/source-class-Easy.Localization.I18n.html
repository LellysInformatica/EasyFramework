143:     /**
144:      * Used by the translation functions in basics.php
145:      * Returns a translated string based on current language and translation files stored in locale folder
146:      *
147:      * @param string $singular String to translate
148:      * @param string $plural Plural string (if any)
149:      * @param string $domain Domain The domain of the translation.
150:      * @param string $category Category The integer value of the category to use.
151:      * @param integer $count Count Count is used with $plural to choose the correct plural form.
152:      * @param string $language Language to translate string to.
153:      *  If null it checks for language in session followed by Config.language configuration variable.
154:      * @return string translated string.
155:      */
156:     public static function translate($singular, $plural = null, $domain = null, $category = 6, $count = null, $language = null)
157:     {
158:         $_this = I18n::getInstance();
159: 
160:         if (strpos($singular, "\r\n") !== false) {
161:             $singular = str_replace("\r\n", "\n", $singular);
162:         }
163:         if ($plural !== null && strpos($plural, "\r\n") !== false) {
164:             $plural = str_replace("\r\n", "\n", $plural);
165:         }
166: 
167:         if (is_numeric($category)) {
168:             $_this->category = $_this->_categories[$category];
169:         }
170: 
171:         $_this->_lang = $_this->loadLanguage($language);
172: 
173:         if (is_null($domain)) {
174:             $domain = self::$defaultDomain;
175:         }
176: 
177:         $_this->domain = $domain . '_' . $_this->l10n->lang;
178:         $cache = new FilesystemCache("tmp/cache");
179: 
180:         if (!isset($_this->_domains[$domain][$_this->_lang])) {
181:             $_this->_domains[$domain][$_this->_lang] = $cache->fetch($_this->domain);
182:         }
183: 
184:         if (!isset($_this->_domains[$domain][$_this->_lang][$_this->category])) {
185:             $_this->_bindTextDomain($domain);
186:             $cache->save($_this->domain, $_this->_domains[$domain][$_this->_lang]);
187:         }
188: 
189:         if ($_this->category == 'LC_TIME') {
190:             return $_this->_translateTime($singular, $domain);
191:         }
192: 
193:         if (!isset($count)) {
194:             $plurals = 0;
195:         } elseif (!empty($_this->_domains[$domain][$_this->_lang][$_this->category]["%plural-c"]) && $_this->_noLocale === false) {
196:             $header = $_this->_domains[$domain][$_this->_lang][$_this->category]["%plural-c"];
197:             $plurals = $_this->_pluralGuess($header, $count);
198:         } else {
199:             if ($count != 1) {
200:                 $plurals = 1;
201:             } else {
202:                 $plurals = 0;
203:             }
204:         }
205: 
206:         if (!empty($_this->_domains[$domain][$_this->_lang][$_this->category][$singular])) {
207:             if (($trans = $_this->_domains[$domain][$_this->_lang][$_this->category][$singular]) || ($plurals) && ($trans = $_this->_domains[$domain][$_this->_lang][$_this->category][$plural])) {
208:                 if (is_array($trans)) {
209:                     if (isset($trans[$plurals])) {
210:                         $trans = $trans[$plurals];
211:                     } else {
212:                         trigger_error(
213:                                 __('Missing plural form translation for "%s" in "%s" domain, "%s" locale. ' .
214:                                         ' Check your po file for correct plurals and valid Plural-Forms header.', $singular, $domain, $_this->_lang
215:                                 ), E_USER_WARNING
216:                         );
217:                         $trans = $trans[0];
218:                     }
219:                 }
220:                 if (strlen($trans)) {
221:                     return $trans;
222:                 }
223:             }
224:         }
225: 
226:         if (!empty($plurals)) {
227:             return $plural;
228:         }
229:         return $singular;
230:     }

254:     /**
255:      * Attempts to find the plural form of a string.
256:      *
257:      * @param string $header Type
258:      * @param integer $n Number
259:      * @return integer plural match
260:      */
261:     protected function _pluralGuess($header, $n)
262:     {
263:         if (!is_string($header) || $header === "nplurals=1;plural=0;" || !isset($header[0])) {
264:             return 0;
265:         }
266: 
267:         if ($header === "nplurals=2;plural=n!=1;") {
268:             return $n != 1 ? 1 : 0;
269:         } elseif ($header === "nplurals=2;plural=n>1;") {
270:             return $n > 1 ? 1 : 0;
271:         }
272: 
273:         if (strpos($header, "plurals=3")) {
274:             if (strpos($header, "100!=11")) {
275:                 if (strpos($header, "10<=4")) {
276:                     return $n % 10 == 1 && $n % 100 != 11 ? 0 : ($n % 10 >= 2 && $n % 10 <= 4 && ($n % 100 < 10 || $n % 100 >= 20) ? 1 : 2);
277:                 } elseif (strpos($header, "100<10")) {
278:                     return $n % 10 == 1 && $n % 100 != 11 ? 0 : ($n % 10 >= 2 && ($n % 100 < 10 || $n % 100 >= 20) ? 1 : 2);
279:                 }
280:                 return $n % 10 == 1 && $n % 100 != 11 ? 0 : ($n != 0 ? 1 : 2);
281:             } elseif (strpos($header, "n==2")) {
282:                 return $n == 1 ? 0 : ($n == 2 ? 1 : 2);
283:             } elseif (strpos($header, "n==0")) {
284:                 return $n == 1 ? 0 : ($n == 0 || ($n % 100 > 0 && $n % 100 < 20) ? 1 : 2);
285:             } elseif (strpos($header, "n>=2")) {
286:                 return $n == 1 ? 0 : ($n >= 2 && $n <= 4 ? 1 : 2);
287:             } elseif (strpos($header, "10>=2")) {
288:                 return $n == 1 ? 0 : ($n % 10 >= 2 && $n % 10 <= 4 && ($n % 100 < 10 || $n % 100 >= 20) ? 1 : 2);
289:             }
290:             return $n % 10 == 1 ? 0 : ($n % 10 == 2 ? 1 : 2);
291:         } elseif (strpos($header, "plurals=4")) {
292:             if (strpos($header, "100==2")) {
293:                 return $n % 100 == 1 ? 0 : ($n % 100 == 2 ? 1 : ($n % 100 == 3 || $n % 100 == 4 ? 2 : 3));
294:             } elseif (strpos($header, "n>=3")) {
295:                 return $n == 1 ? 0 : ($n == 2 ? 1 : ($n == 0 || ($n >= 3 && $n <= 10) ? 2 : 3));
296:             } elseif (strpos($header, "100>=1")) {
297:                 return $n == 1 ? 0 : ($n == 0 || ($n % 100 >= 1 && $n % 100 <= 10) ? 1 : ($n % 100 >= 11 && $n % 100 <= 20 ? 2 : 3));
298:             }
299:         } elseif (strpos($header, "plurals=5")) {
300:             return $n == 1 ? 0 : ($n == 2 ? 1 : ($n >= 3 && $n <= 6 ? 2 : ($n >= 7 && $n <= 10 ? 3 : 4)));
301:         }
302:     }

304:     /**
305:      * Binds the given domain to a file in the specified directory.
306:      *
307:      * @param string $domain Domain to bind
308:      * @return string Domain binded
309:      */
310:     protected function _bindTextDomain($domain)
311:     {
312:         $this->_noLocale = true;
313:         $core = true;
314:         $merge = array();
315:         $searchPaths = array(APP_PATH . "Locale");
316: 
317:         foreach ($searchPaths as $directory) {
318:             foreach ($this->l10n->languagePath as $lang) {
319:                 $localeDef = $directory . $lang . DS . $this->category;
320:                 if (is_file($localeDef)) {
321:                     $definitions = self::loadLocaleDefinition($localeDef);
322:                     if ($definitions !== false) {
323:                         $this->_domains[$domain][$this->_lang][$this->category] = self::loadLocaleDefinition($localeDef);
324:                         $this->_noLocale = false;
325:                         return $domain;
326:                     }
327:                 }
328: 
329:                 if ($core) {
330:                     $app = $directory . $lang . DS . $this->category . DS . 'core';
331:                     $translations = false;
332: 
333:                     if (is_file($app . '.mo')) {
334:                         $translations = self::loadMo($app . '.mo');
335:                     }
336:                     if ($translations === false && is_file($app . '.po')) {
337:                         $translations = self::loadPo($app . '.po');
338:                     }
339: 
340:                     if ($translations !== false) {
341:                         $this->_domains[$domain][$this->_lang][$this->category] = $translations;
342:                         $merge[$domain][$this->_lang][$this->category] = $this->_domains[$domain][$this->_lang][$this->category];
343:                         $this->_noLocale = false;
344:                         $core = null;
345:                     }
346:                 }
347: 
348:                 $file = $directory . $lang . DS . $this->category . DS . $domain;
349:                 $translations = false;
350: 
351:                 if (is_file($file . '.mo')) {
352:                     $translations = self::loadMo($file . '.mo');
353:                 }
354:                 if ($translations === false && is_file($file . '.po')) {
355:                     $translations = self::loadPo($file . '.po');
356:                 }
357: 
358:                 if ($translations !== false) {
359:                     $this->_domains[$domain][$this->_lang][$this->category] = $translations;
360:                     $this->_noLocale = false;
361:                     break 2;
362:                 }
363:             }
364:         }
365: 
366:         if (empty($this->_domains[$domain][$this->_lang][$this->category])) {
367:             $this->_domains[$domain][$this->_lang][$this->category] = array();
368:             return $domain;
369:         }
370: 
371:         if (isset($this->_domains[$domain][$this->_lang][$this->category][""])) {
372:             $head = $this->_domains[$domain][$this->_lang][$this->category][""];
373: 
374:             foreach (explode("\n", $head) as $line) {
375:                 $header = strtok($line, ":");
376:                 $line = trim(strtok("\n"));
377:                 $this->_domains[$domain][$this->_lang][$this->category]["%po-header"][strtolower($header)] = $line;
378:             }
379: 
380:             if (isset($this->_domains[$domain][$this->_lang][$this->category]["%po-header"]["plural-forms"])) {
381:                 $switch = preg_replace("/(?:[() {}\\[\\]^\\s*\\]]+)/", "", $this->_domains[$domain][$this->_lang][$this->category]["%po-header"]["plural-forms"]);
382:                 $this->_domains[$domain][$this->_lang][$this->category]["%plural-c"] = $switch;
383:                 unset($this->_domains[$domain][$this->_lang][$this->category]["%po-header"]);
384:             }
385:             $this->_domains = Hash::mergeDiff($this->_domains, $merge);
386: 
387:             if (isset($this->_domains[$domain][$this->_lang][$this->category][null])) {
388:                 unset($this->_domains[$domain][$this->_lang][$this->category][null]);
389:             }
390:         }
391: 
392:         return $domain;
393:     }

395:     /**
396:      * Loads the binary .mo file and returns array of translations
397:      *
398:      * @param string $filename Binary .mo file to load
399:      * @return mixed Array of translations on success or false on failure
400:      */
401:     public static function loadMo($filename)
402:     {
403:         $translations = false;
404: 
405:         if ($data = file_get_contents($filename)) {
406:             $translations = array();
407:             $header = substr($data, 0, 20);
408:             $header = unpack("L1magic/L1version/L1count/L1o_msg/L1o_trn", $header);
409:             extract($header);
410: 
411:             if ((dechex($magic) == '950412de' || dechex($magic) == 'ffffffff950412de') && $version == 0) {
412:                 for ($n = 0; $n < $count; $n++) {
413:                     $r = unpack("L1len/L1offs", substr($data, $o_msg + $n * 8, 8));
414:                     $msgid = substr($data, $r["offs"], $r["len"]);
415:                     unset($msgid_plural);
416: 
417:                     if (strpos($msgid, "\000")) {
418:                         list($msgid, $msgid_plural) = explode("\000", $msgid);
419:                     }
420:                     $r = unpack("L1len/L1offs", substr($data, $o_trn + $n * 8, 8));
421:                     $msgstr = substr($data, $r["offs"], $r["len"]);
422: 
423:                     if (strpos($msgstr, "\000")) {
424:                         $msgstr = explode("\000", $msgstr);
425:                     }
426:                     $translations[$msgid] = $msgstr;
427: 
428:                     if (isset($msgid_plural)) {
429:                         $translations[$msgid_plural] = & $translations[$msgid];
430:                     }
431:                 }
432:             }
433:         }
434: 
435:         return $translations;
436:     }

438:     /**
439:      * Loads the text .po file and returns array of translations
440:      *
441:      * @param string $filename Text .po file to load
442:      * @return mixed Array of translations on success or false on failure
443:      */
444:     public static function loadPo($filename)
445:     {
446:         if (!$file = fopen($filename, "r")) {
447:             return false;
448:         }
449: 
450:         $type = 0;
451:         $translations = array();
452:         $translationKey = "";
453:         $plural = 0;
454:         $header = "";
455: 
456:         do {
457:             $line = trim(fgets($file));
458:             if ($line == "" || $line[0] == "#") {
459:                 continue;
460:             }
461:             if (preg_match("/msgid[[:space:]]+\"(.+)\"$/i", $line, $regs)) {
462:                 $type = 1;
463:                 $translationKey = stripcslashes($regs[1]);
464:             } elseif (preg_match("/msgid[[:space:]]+\"\"$/i", $line, $regs)) {
465:                 $type = 2;
466:                 $translationKey = "";
467:             } elseif (preg_match("/^\"(.*)\"$/i", $line, $regs) && ($type == 1 || $type == 2 || $type == 3)) {
468:                 $type = 3;
469:                 $translationKey .= stripcslashes($regs[1]);
470:             } elseif (preg_match("/msgstr[[:space:]]+\"(.+)\"$/i", $line, $regs) && ($type == 1 || $type == 3) && $translationKey) {
471:                 $translations[$translationKey] = stripcslashes($regs[1]);
472:                 $type = 4;
473:             } elseif (preg_match("/msgstr[[:space:]]+\"\"$/i", $line, $regs) && ($type == 1 || $type == 3) && $translationKey) {
474:                 $type = 4;
475:                 $translations[$translationKey] = "";
476:             } elseif (preg_match("/^\"(.*)\"$/i", $line, $regs) && $type == 4 && $translationKey) {
477:                 $translations[$translationKey] .= stripcslashes($regs[1]);
478:             } elseif (preg_match("/msgid_plural[[:space:]]+\".*\"$/i", $line, $regs)) {
479:                 $type = 6;
480:             } elseif (preg_match("/^\"(.*)\"$/i", $line, $regs) && $type == 6 && $translationKey) {
481:                 $type = 6;
482:             } elseif (preg_match("/msgstr\[(\d+)\][[:space:]]+\"(.+)\"$/i", $line, $regs) && ($type == 6 || $type == 7) && $translationKey) {
483:                 $plural = $regs[1];
484:                 $translations[$translationKey][$plural] = stripcslashes($regs[2]);
485:                 $type = 7;
486:             } elseif (preg_match("/msgstr\[(\d+)\][[:space:]]+\"\"$/i", $line, $regs) && ($type == 6 || $type == 7) && $translationKey) {
487:                 $plural = $regs[1];
488:                 $translations[$translationKey][$plural] = "";
489:                 $type = 7;
490:             } elseif (preg_match("/^\"(.*)\"$/i", $line, $regs) && $type == 7 && $translationKey) {
491:                 $translations[$translationKey][$plural] .= stripcslashes($regs[1]);
492:             } elseif (preg_match("/msgstr[[:space:]]+\"(.+)\"$/i", $line, $regs) && $type == 2 && !$translationKey) {
493:                 $header .= stripcslashes($regs[1]);
494:                 $type = 5;
495:             } elseif (preg_match("/msgstr[[:space:]]+\"\"$/i", $line, $regs) && !$translationKey) {
496:                 $header = "";
497:                 $type = 5;
498:             } elseif (preg_match("/^\"(.*)\"$/i", $line, $regs) && $type == 5) {
499:                 $header .= stripcslashes($regs[1]);
500:             } else {
501:                 unset($translations[$translationKey]);
502:                 $type = 0;
503:                 $translationKey = "";
504:                 $plural = 0;
505:             }
506:         } while (!feof($file));
507:         fclose($file);
508: 
509:         $merge[""] = $header;
510:         return array_merge($merge, $translations);
511:     }

513:     /**
514:      * Parses a locale definition file following the POSIX standard
515:      *
516:      * @param string $filename Locale definition filename
517:      * @return mixed Array of definitions on success or false on failure
518:      */
519:     public static function loadLocaleDefinition($filename)
520:     {
521:         if (!$file = fopen($filename, "r")) {
522:             return false;
523:         }
524: 
525:         $definitions = array();
526:         $comment = '#';
527:         $escape = '\\';
528:         $currentToken = false;
529:         $value = '';
530:         $_this = I18n::getInstance();
531:         while ($line = fgets($file)) {
532:             $line = trim($line);
533:             if (empty($line) || $line[0] === $comment) {
534:                 continue;
535:             }
536:             $parts = preg_split("/[[:space:]]+/", $line);
537:             if ($parts[0] === 'comment_char') {
538:                 $comment = $parts[1];
539:                 continue;
540:             }
541:             if ($parts[0] === 'escape_char') {
542:                 $escape = $parts[1];
543:                 continue;
544:             }
545:             $count = count($parts);
546:             if ($count == 2) {
547:                 $currentToken = $parts[0];
548:                 $value = $parts[1];
549:             } elseif ($count == 1) {
550:                 $value .= $parts[0];
551:             } else {
552:                 continue;
553:             }
