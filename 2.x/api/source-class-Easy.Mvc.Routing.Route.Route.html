182:     /**
183:      * Builds a route regular expression.  Uses the template, defaults and options
184:      * properties to compile a regular expression that can be used to parse request strings.
185:      *
186:      * @return void
187:      */
188:     protected function _writeRoute()
189:     {
190:         if (empty($this->template) || ($this->template === '/')) {
191:             $this->_compiledRoute = '#^/*$#';
192:             $this->keys = array();
193:             return;
194:         }
195:         $route = $this->template;
196:         $names = $routeParams = array();
197:         $parsed = preg_quote($this->template, '#');
198: 
199:         preg_match_all('#:([A-Za-z0-9_-]+[A-Z0-9a-z])#', $route, $namedElements);
200:         foreach ($namedElements[1] as $i => $name) {
201:             $search = '\\' . $namedElements[0][$i];
202:             if (isset($this->options[$name])) {
203:                 $option = null;
204:                 if ($name !== 'plugin' && array_key_exists($name, $this->defaults)) {
205:                     $option = '?';
206:                 }
207:                 $slashParam = '/\\' . $namedElements[0][$i];
208:                 if (strpos($parsed, $slashParam) !== false) {
209:                     $routeParams[$slashParam] = '(?:/(?P<' . $name . '>' . $this->options[$name] . ')' . $option . ')' . $option;
210:                 } else {
211:                     $routeParams[$search] = '(?:(?P<' . $name . '>' . $this->options[$name] . ')' . $option . ')' . $option;
212:                 }
213:             } else {
214:                 $routeParams[$search] = '(?:(?P<' . $name . '>[^/]+))';
215:             }
216:             $names[] = $name;
217:         }
218:         if (preg_match('#\/\*\*$#', $route)) {
219:             $parsed = preg_replace('#/\\\\\*\\\\\*$#', '(?:/(?P<_trailing_>.*))?', $parsed);
220:             $this->_greedy = true;
221:         }
222:         if (preg_match('#\/\*$#', $route)) {
223:             $parsed = preg_replace('#/\\\\\*$#', '(?:/(?P<_args_>.*))?', $parsed);
224:             $this->_greedy = true;
225:         }
226:         krsort($routeParams);
227:         $parsed = str_replace(array_keys($routeParams), array_values($routeParams), $parsed);
228:         $this->_compiledRoute = '#^' . $parsed . '[/]*$#';
229:         $this->keys = $names;
230: 
231:         // Remove defaults that are also keys. They can cause match failures
232:         foreach ($this->keys as $key) {
233:             unset($this->defaults[$key]);
234:         }
235:     }

270:     /**
271:      * Checks to see if the given URL can be parsed by this route.
272:      * If the route can be parsed an array of parameters will be returned; if not
273:      * false will be returned. String urls are parsed if they match a routes regular expression.
274:      *
275:      * @param string $url The url to attempt to parse.
276:      * @return mixed Boolean false on failure, otherwise an array or parameters
277:      */
278:     public function parse($url)
279:     {
280: 
281:         if (!$this->compiled()) {
282:             $this->compile();
283:         }
284:         list($url, $ext) = $this->_parseExtension($url);
285: 
286:         if (!preg_match($this->_compiledRoute, $url, $route)) {
287:             return false;
288:         }
289:         foreach ($this->defaults as $key => $val) {
290:             $key = (string) $key;
291:             if ($key[0] === '[' && preg_match('/^\[(\w+)\]$/', $key, $header)) {
292:                 if (isset($this->_headerMap[$header[1]])) {
293:                     $header = $this->_headerMap[$header[1]];
294:                 } else {
295:                     $header = 'http_' . $header[1];
296:                 }
297:                 $header = strtoupper($header);
298: 
299:                 $val = (array) $val;
300:                 $h = false;
301: 
302:                 foreach ($val as $v) {
303:                     if ($this->context->server->contains($header)) {
304:                         $value = $this->context->server->getItem($header);
305:                         if ($value === $v) {
306:                             $h = true;
307:                         }
308:                     }
309:                 }
310:                 if (!$h) {
311:                     return false;
312:                 }
313:             }
314:         }
315:         array_shift($route);
316:         $count = count($this->keys);
317:         for ($i = 0; $i <= $count; $i++) {
318:             unset($route[$i]);
319:         }
320:         $route['pass'] = array();
321: 
322:         // Assign defaults, set passed args to pass
323:         foreach ($this->defaults as $key => $value) {
324:             if (isset($route[$key])) {
325:                 continue;
326:             }
327:             if (is_int($key)) {
328:                 $route['pass'][] = $value;
329:                 continue;
330:             }
331:             $route[$key] = $value;
332:         }
333: 
334:         foreach ($this->keys as $key) {
335:             if (isset($route[$key])) {
336:                 $route[$key] = rawurldecode($route[$key]);
337:             }
338:         }
339: 
340:         if (isset($route['_args_'])) {
341:             $pass = $this->_parseArgs($route['_args_'], $route);
342:             $route['pass'] = array_merge($route['pass'], $pass);
343:             unset($route['_args_']);
344:         }
345: 
346:         if (isset($route['_trailing_'])) {
347:             $route['pass'][] = rawurldecode($route['_trailing_']);
348:             unset($route['_trailing_']);
349:         }
350: 
351:         if ($ext && empty($route['_ext'])) {
352:             $route['_ext'] = $ext;
353:         }
354: 
355:         // restructure 'pass' key route params
356:         if (isset($this->options['pass'])) {
357:             $j = count($this->options['pass']);
358:             while ($j--) {
359:                 if (isset($route[$this->options['pass'][$j]])) {
360:                     array_unshift($route['pass'], $route[$this->options['pass'][$j]]);
361:                 }
362:             }
363:         }
364:         return $route;
365:     }

367:     /**
368:      * Removes the extension if the $url contains a known extension.
369:      * If there are no known extensions all extensions are supported.
370:      *
371:      * @param string $url The url to parse.
372:      * @return array containing url, extension
373:      */
374:     protected function _parseExtension($url)
375:     {
376:         if (empty($this->_extensions)) {
377:             return array($url, null);
378:         }
379:         preg_match('/\.([0-9a-z]*)$/', $url, $match);
380:         if (empty($match[1])) {
381:             return array($url, null);
382:         }
383:         $ext = strtolower($match[1]);
384:         $len = strlen($match[1]);
385:         foreach ($this->_extensions as $name) {
386:             if (strtolower($name) === $ext) {
387:                 $url = substr($url, 0, ($len + 1) * -1);
388:                 return array($url, $ext);
389:             }
390:         }
391:         return array($url, null);
392:     }

395:      * Parse passed parameters into a list of passed args.
396:      *
397:      * @param string $args A string with the passed params.  eg. /1/foo
398:      * @param string $context The current route context, which should contain controller/action keys.
399:      * @return array Array of passed args.
400:      */
401:     protected function _parseArgs($args, $context)
402:     {
403:         $pass = array();
404:         $args = explode('/', $args);
405: 
406:         foreach ($args as $param) {
407:             if (empty($param) && $param !== '0' && $param !== 0) {
408:                 continue;
409:             }
410:             $pass[] = rawurldecode($param);

415:     /**
416:      * Attempt to match a url array.  If the url matches the route parameters and settings, then
417:      * return a generated string url.  If the url doesn't match the route parameters, false will be returned.
418:      * This method handles the reverse routing or conversion of url arrays into string urls.
419:      *
420:      * @param array $url An array of parameters to check matching with.
421:      * @param array $context An array of the current request context.
422:      *   Contains information such as the current host, scheme, port, and base
423:      *   directory.
424:      * @return mixed Either a string url for the parameters if they match or false.
425:      */
426:     public function match($url, $context = array())
427:     {
428:         if (!$this->compiled()) {
429:             $this->compile();
430:         }
431:         $defaults = $this->defaults;
432: 
433:         $hostOptions = array_intersect_key($url, $context);
434: 
435:         // Check for properties that will cause and
436:         // absoulte url. Copy the other properties over.
437:         if (
438:                 isset($hostOptions['_scheme']) ||
439:                 isset($hostOptions['_port']) ||
440:                 isset($hostOptions['_host'])
441:         ) {
442:             $hostOptions += $context;
443: 
444:             if ($hostOptions['_port'] == $context['_port']) {
445:                 unset($hostOptions['_port']);
446:             }
447:         }
448: 
449:         // If no base is set, copy one in.
450:         if (!isset($hostOptions['_base']) && isset($context['_base'])) {
451:             $hostOptions['_base'] = $context['_base'];
452:         }
453:         unset($url['_host'], $url['_scheme'], $url['_port'], $url['_base']);
454: 
455:         // Move extension into the hostOptions so its not part of
456:         // reverse matches.
457:         if (isset($url['_ext'])) {
458:             $hostOptions['_ext'] = $url['_ext'];
459:             unset($url['_ext']);
460:         }
461: 
462:         // Missing defaults is a fail.
463:         if (array_diff_key($defaults, $url) !== array()) {
464:             return false;
465:         }
466: 
467:         // Defaults with different values are a fail.
468:         if (array_intersect_key($url, $defaults) != $defaults) {
469:             return false;
470:         }
471: 
472:         // check that all the key names are in the url
473:         $keyNames = array_flip($this->keys);
474:         if (array_intersect_key($keyNames, $url) !== $keyNames) {
475:             return false;
476:         }
477: 
478:         $prefixes = Mapper::getPrefixes();
479:         $pass = array();
480:         $query = array();
481: 
482:         foreach ($url as $key => $value) {
483:             // keys that exist in the defaults and have different values is a match failure.
484:             $defaultExists = array_key_exists($key, $defaults);
485: 
486:             // If the key is a routed key, its not different yet.
487:             if (array_key_exists($key, $keyNames)) {
488:                 continue;
489:             }
490: 
491:             // pull out passed args
492:             $numeric = is_numeric($key);
493:             if ($numeric && isset($defaults[$key]) && $defaults[$key] == $value) {
494:                 continue;
495:             } elseif ($numeric) {
496:                 //$pass[] = $value;
497:                 $pass = $value;
498:                 unset($url[$key]);
499:                 continue;
500:             }
501: 
502:             // keys that don't exist are different.
503:             if (!$defaultExists && ($value !== null && $value !== false && $value !== '')) {
504:                 $query[$key] = $value;
505:                 unset($url[$key]);
506:             }
507:         }
508: 
509:         // if not a greedy route, no extra params are allowed.
510:         if (!$this->_greedy && !empty($pass)) {
511:             return false;
512:         }
513: 
514:         //check patterns for routed params
515:         if (!empty($this->options)) {
516:             foreach ($this->options as $key => $pattern) {
517:                 if (isset($url[$key]) && !preg_match('#^' . $pattern . '$#', $url[$key])) {
518:                     return false;
519:                 }
520:             }
521:         }
522:         $url += $hostOptions;
523:         return $this->_writeUrl($url, $pass, $query);
524:     }

526:     /**
527:      * Converts a matching route array into a url string. Composes the string url using the template
528:      * used to create the route.
529:      *
530:      * @param array $params The params to convert to a string url.
531:      * @param array $pass The additional passed arguments.
532:      * @return string Composed route string.
533:      */
534:     protected function _writeUrl($params, $pass = array(), $query = array())
535:     {
536:         $pass = implode('/', (array) $pass);
537:         $out = $this->template;
538: 
539:         $search = $replace = array();
540:         foreach ($this->keys as $key) {
541:             $string = null;
542:             if (isset($params[$key])) {
543:                 $string = $params[$key];
544:             } elseif (strpos($out, $key) != strlen($out) - strlen($key)) {
545:                 $key .= '/';
546:             }
547:             $search[] = ':' . $key;
548:             $replace[] = $string;
549:         }
550:         $out = str_replace($search, $replace, $out);
551: 
552:         if (strpos($this->template, '*')) {
553:             $out = str_replace('*', $pass, $out);
554:         }
555: 
556:         // add base url if applicable.
557:         if (isset($params['_base'])) {
558:             $out = $params['_base'] . $out;
559:             unset($params['_base']);
560:         }
561: 
562:         $out = str_replace('//', '/', $out);
563: 
564:         if (
565:                 isset($params['_scheme']) ||
566:                 isset($params['_host']) ||
567:                 isset($params['_port'])
568:         ) {
569:             $host = $params['_host'];
570: 
571:             // append the port if it exists.
572:             if (isset($params['_port'])) {
573:                 $host .= ':' . $params['_port'];
574:             }
575:             $out = sprintf(
576:                     '%s://%s%s', $params['_scheme'], $host, $out
577:             );
578:         }
579:         if (!empty($params['_ext']) || !empty($query)) {
580:             $out = rtrim($out, '/');
581:         }
582:         if (!empty($params['_ext'])) {
583:             $out .= '.' . $params['_ext'];
584:         }
585:         if (!empty($query)) {
586:             $out .= '?' . http_build_query($query);
587:         }
588:         return $out;
589:     }
