 398:      *

 400:      */

 402:     {
 403:         if ($this->isInformational() || in_array($this->statusCode, array(204, 304))) {
 404:             $this->setContent(null);
 405:         }

 407:         // Content-type based on the Request
 408:         if (!$this->headers->contains('Content-Type')) {
 409:             $format = $request->getRequestFormat();
 410:             if (null !== $format && $mimeType = $request->getMimeType($format)) {
 411:                 $this->headers->set('Content-Type', $mimeType);
 412:             }
 413:         }
 414:

 416:         if (strpos($this->_contentType, 'text/') === 0) {

 450:     /**
 451:      * Sends the complete response to the client including headers and message body.
 452:      * Will echo out the content in the response body.
 453:      *
 454:      * @return void
 455:      */
 456:     public function send()
 457:     {
 458:         $this->sendHeaders();
 459:         $this->sendContent($this->content);
 460: 
 461:         if (function_exists('fastcgi_finish_request')) {
 462:             fastcgi_finish_request();
 463:         } elseif ('cli' !== PHP_SAPI) {
 464:             // ob_get_level() never returns 0 on some Windows configurations, so if
 465:             // the level is the same two times in a row, the loop should be stopped.
 466:             $previous = null;
 467:             $obStatus = ob_get_status(1);
 468:             while (($level = ob_get_level()) > 0 && $level !== $previous) {
 469:                 $previous = $level;
 470:                 if ($obStatus[$level - 1] && isset($obStatus[$level - 1]['del']) && $obStatus[$level - 1]['del']) {
 471:                     ob_end_flush();
 472:                 }
 473:             }
 474:             flush();
 475:         }
 476: 
 477:         return $this;
 478:     }

 480:     /**
 481:      * Sends HTTP headers.
 482:      *
 483:      * @return Response
 484:      */
 485:     protected function sendHeaders()
 486:     {
 487:         // headers have already been sent by the developer
 488:         if (headers_sent()) {
 489:             return $this;
 490:         }
 491:         // status
 492:         header(sprintf('HTTP/%s %s %s', $this->protocol, $this->statusCode, $this->statusText));
 493:         // headers
 494:         foreach ($this->headers as $header => $value) {
 495:             header("{$header}: {$value}");
 496:         }
 497: 
 498:         // cookies
 499:         foreach ($this->headers->getCookies() as $cookie) {
 500:             setcookie($cookie->getName(), $cookie->getValue(), $cookie->getExpiresTime(), $cookie->getPath(), $cookie->getDomain(), $cookie->isSecure(), $cookie->isHttpOnly());
 501:         }
 502:     }

 603:      *

 605:      *
 606:      * ### Returning the current content type
 607:      *
 608:      * e.g `type();`
 609:      *
 610:      * ### Storing a content type definition
 611:      *
 612:      * e.g `type(array('keynote' => 'application/keynote'));`

 797:     /**

 799:      * This method controls the `public` or `private` directive in the Cache-Control
 800:      * header

 802:      * @param boolean $public  if set to true, the Cache-Control header will be set as public
 803:      * if set to false, the response will be set to private
 804:      * if no value is provided, it will return whether the response is sharable or not
 805:      * @param int $time time in seconds after which the response should no longer be considered fresh

 807:      */

 809:     {
 810:         if ($public === null) {
