 26: /**
 27:  * Session Helper.
 28:  *
 29:  * Session reading from the view.
 30:  *
 31:  * @package       Easy.View.Helper
 32:  */
 33: class SessionHelper extends AppHelper
 34: {
 35: 
 36:     /**
 37:      * Used to read a session values set in a controller for a key or return values for all keys.
 38:      *
 39:      * In your view: `$this->Session->read('Controller.sessKey');`
 40:      * Calling the method without a param will return all session vars
 41:      *
 42:      * @param string $name the name of the session key you want to read
 43:      * @return mixed values from the session vars
 44:      */
 45:     public function read($name = null)
 46:     {
 47:         return Session::read($name);
 48:     }
 49: 
 50:     /**
 51:      * Returns last error encountered in a session
 52:      *
 53:      * In your view: `$this->Session->error();`
 54:      *
 55:      * @return string last error
 56:      */
 57:     public function error()
 58:     {
 59:         return Session::error();
 60:     }
 61: 
 62:     /**
 63:      * Used to check is a session key has been set
 64:      *
 65:      * In your view: `$this->Session->check('Controller.sessKey');`
 66:      *
 67:      * @param string $name
 68:      * @return boolean
 69:      */
 70:     public function check($name)
 71:     {
 72:         return Session::check($name);
 73:     }
 74: 
 75:     /**
 76:      * Used to render the message set in Controller::Session::setFlash()
 77:      *
 78:      * In your view: $this->Session->flash('somekey');
 79:      * Will default to flash if no param is passed
 80:      *
 81:      * You can pass additional information into the flash message generation.  This allows you
 82:      * to consolidate all the parameters for a given type of flash message into the view.
 83:      *
 84:      * {{{
 85:      * echo $this->Session->flash('flash', array('params' => array('class' => 'new-flash')));
 86:      * }}}
 87:      *
 88:      * The above would generate a flash message with a custom class name. Using $attrs['params'] you
 89:      * can pass additional data into the element rendering that will be made available as local variables
 90:      * when the element is rendered:
 91:      *
 92:      * {{{
 93:      * echo $this->Session->flash('flash', array('params' => array('name' => $user['User']['name'])));
 94:      * }}}
 95:      *
 96:      * This would pass the current user's name into the flash message, so you could create peronsonalized
 97:      * messages without the controller needing access to that data.
 98:      *
 99:      * Lastly you can choose the element that is rendered when creating the flash message. Using
100:      * custom elements allows you to fully customize how flash messages are generated.
101:      *
102:      * {{{
103:      * echo $this->Session->flash('flash', array('element' => 'my_custom_element'));
104:      * }}}
105:      *
106:      * If you want to use an element from a plugin for rendering your flash message you can do that using the 
107:      * plugin param:
108:      *
109:      * {{{
110:      * echo $this->Session->flash('flash', array(
111:      *      'element' => 'my_custom_element',
112:      *      'params' => array('plugin' => 'my_plugin')
113:      * ));
114:      * }}}
115:      *
116:      * @param string $key The [Message.]key you are rendering in the view.
117:      * @param array $attrs Additional attributes to use for the creation of this flash message.
118:      *    Supports the 'params', and 'element' keys that are used in the helper.
119:      * @return string
120:      */
121:     public function flash($key = 'flash', array $attr = array())
122:     {
123:         $out = false;
124:         $attr = Hash::merge(array(
125:                     'class' => null,
126:                     'tag' => false
127:                         ), $attr
128:         );
129: 
130:         if (Session::check('Message.' . $key)) {
131:             $message = Session::read('Message.' . $key);
132: 
133:             if ($attr['tag']) {
134:                 if (!is_array($message)) {
135:                     $message = array($message);
136:                 }
137: 
138:                 $out = "<div class='{$attr['class']}'>";
139:                 foreach ($message as $value) {
140:                     $out .= "<p>" . $value . "</p>";
141:                 };
142:                 $out .= "</div>";
143:             } else {
144:                 $out = $message;
145:             }
146:         }
147: 
148:         Session::delete('Message.' . $key);
149: 
150:         return $out;
151:     }
152: 
153: }
