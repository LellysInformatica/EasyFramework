342:     {
343:         if ($controller === true) {

346:         $this->eventDispatcher->dispatch("beforeRender", new ShutdownEvent($this));
347:         $view = new View($this, $this->container->get("Templating"));
348:         //Pass the view vars to view class
349:         foreach ($this->viewVars as $key => $value) {
350:             $view->set($key, $value);
351:         }
352: 
353:         $content = $view->display("{$controller}/{$action}", $layout, null, $output);
354: 
355:         //We set the autorender to false, this prevent the action to call this method 2 times
356:         $this->setAutoRender(false);
357: 
358:         if ($output === true) {
359:             $response = new Response();
360:             // Display the view
361:             $response->setContent($content);
362:             return $response;
363:         } else {
364:             return $content;
365:         }
